33  *
34  * Sums up sizes of all list elements only, length itself is not included.
35  * Therefore, the returned length is suitable as argument for serialize_length.
36  *
37  * \tparam T list element type
38  * \param list
39  * \return accumulated elements' size
40  */

43 {

45  for (auto& elem : list) {
46  size += get_size(elem);
47  }
48  return size;

78 /** \brief Deserialize a list from given archive
79  * \tparam T the type of the list
80  * \tparam ARGS all additional arguments for the underlying functions
81  * \param ar, shall start with the list
82  * \param args the additional arguments
83  * \return size of the deserialized list in bytes
84  */
85 template<class T, typename... ARGS>
86 std::size_t deserialize(InputArchive& ar, std::list<T>& list, ARGS&&... args)
87 {
88  const auto length = trim_size(deserialize_length(ar));
89  std::intmax_t remainder = length;
90  while (remainder > 0) {
91  T t;
92  remainder -= deserialize(ar, t, std::forward<ARGS>(args)...);
93  list.push_back(std::move(t));
94  }
95  assert(remainder == 0);
96  return length;
97 }
